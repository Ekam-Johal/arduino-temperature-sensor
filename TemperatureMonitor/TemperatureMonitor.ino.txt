#include <Wire.h>
#include <LiquidCrystal_I2C.h>
#include <DHT.h>

// --------- DHT11 CONFIG ----------
#define DHTPIN 2
#define DHTTYPE DHT11
DHT dht(DHTPIN, DHTTYPE);

// --------- LCD CONFIG ------------
// If nothing shows on screen, try 0x3F instead of 0x27
LiquidCrystal_I2C lcd(0x27, 16, 2);

void setup() {
  lcd.init();
  lcd.backlight();

  dht.begin();

  lcd.setCursor(0, 0);
  lcd.print("Temp Monitor");
  lcd.setCursor(0, 1);
  lcd.print("Starting...");
  delay(1500);
}

void loop() {
  float temp = dht.readTemperature();   // Celsius
  float hum  = dht.readHumidity();      // %

  lcd.clear();

  lcd.setCursor(0, 0);
  lcd.print("Temp: ");
  if (isnan(temp)) {
    lcd.print("ERR");
  } else {
    lcd.print(temp, 1);
    lcd.print(" C");
  }

  lcd.setCursor(0, 1);
  lcd.print("Hum:  ");
  if (isnan(hum)) {
    lcd.print("ERR");
  } else {
    lcd.print(hum, 1);
    lcd.print(" %");
  }

  delay(2000);
}
